import random

actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3

def choose_action(epsilon):
    
    action = None
    if random.uniform(0,1) < epsilon: # 
        # 
        i = int(random.uniform(0,4))
        action = actions[i]
    else: 
        #  Q 
        action = max(qline, key=lambda k: qline[k])
    return action
